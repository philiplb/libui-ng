517// uiAttribute stores information about an attribute in a
518// uiAttributedString.
519//
520// You do not create uiAttributes directly; instead, you create a
521// uiAttribute of a given type using the specialized constructor
522// functions. For every Unicode codepoint in the uiAttributedString,
523// at most one value of each attribute type can be applied.
524//
525// uiAttributes are immutable and the uiAttributedString takes
526// ownership of the uiAttribute object once assigned, copying its
527// contents as necessary.
528typedef struct uiAttribute uiAttribute;
529
530// @role uiAttribute destructor
531// uiFreeAttribute() frees a uiAttribute. You generally do not need to
532// call this yourself, as uiAttributedString does this for you. In fact,
